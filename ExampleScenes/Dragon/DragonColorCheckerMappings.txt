    FloorMaterial-material:material:matte

    SunX-light:intensity.spectrum = D65.spd
    SunY-light:light:directional
    SunY-light:intensity.spectrum = D65.spd
    SunZ-light:light:directional
    SunZ-light:intensity.spectrum = D65.spd
